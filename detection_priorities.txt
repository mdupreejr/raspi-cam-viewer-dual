# Frigate VLC Viewer - Detection Priority Configuration
# 
# Configure which detection types to show and their priorities.
# Higher priority detections will interrupt lower priority ones.
# Priority levels: 1-10 (10 = highest priority)
# Set to 0 to ignore that detection type completely
#
# Format: detection_type=priority

# People and animals (high priority)
person=10
dog=8
cat=7

# Vehicles (medium priority)  
car=5
motorcycle=5
bus=4
truck=4

# Other objects (low priority)
bicycle=3
package=3

# Set to 0 to ignore
bird=0
